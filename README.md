## Installation
```
npm install --save @angular-mdc/web
```

## Getting started
### Include all components
```ts
import { MaterialModule } from '@angular-mdc/web';
```
### Include individual components
```ts
import { FabModule, MenuModule } from '@angular-mdc/web';
```
